DROP TABLE IF EXISTS user;

CREATE TABLE user (
    id_user SERIAL PRIMARY KEY,
    nome VARCHAR(70) NOT NULL,
    ra VARCHAR(7) UNIQUE,
    login VARCHAR(20) UNIQUE NOT NULL,
    senha TEXT NOT NULL,
    ano INT,
    tipo VARCHAR(15) NOT NULL,
    telefone VARCHAR(15),
    bloqueado BOOLEAN NOT NULL DEFAULT FALSE
);
